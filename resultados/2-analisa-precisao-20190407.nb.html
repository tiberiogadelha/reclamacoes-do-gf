normaliza = function(lexico, polaridade) {
    if(lexico == "sentimento_op30"){
        return ((((polaridade - minimo_op30)/(maximo_op30 - minimo_op30)) * -4) + 5)
    } else {
        return ((((polaridade - minimo_sent)/(maximo_sent - minimo_sent)) * -4) + 5)
    }
}
